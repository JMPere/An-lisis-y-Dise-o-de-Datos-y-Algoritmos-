public static void tiempoMinimo(Graph<Monumento,Camino> grafo,List<Monumento> monumentos) {
		Integer i = 0, j = 0;
		Boolean hayCamino = true;
		GraphPath<Monumento,Camino> dijkstraCamino = null;
		while(i<monumentos.size()-1) {
			DijkstraShortestPath<Monumento,Camino> dijkstra = new DijkstraShortestPath<Monumento,Camino>(grafo);
			dijkstraCamino = dijkstra.getPath(monumentos.get(i), monumentos.get(i+1));
			if(dijkstraCamino == null) {
				System.out.println("No es posible visitar los monumentos en el orden dado");
				hayCamino = false;
				break;
				}
			i++;
			}
			if(hayCamino) {
				System.out.println("El camino minimo es:");
				List<Monumento> trayecto = dijkstraCamino.getVertexList();
				while(j < trayecto.size()-1) {
					System.out.println(trayecto.get(j)+"->");
					j++;
				}
				System.out.println(trayecto.get(trayecto.size()));
			}
		}
